.section ".data"					//Defines data section for hardware registers
.equ	DDRB,0x04					//Sets PB0-PB7 as I/O pins (Port B)
.equ	DDRD,0x0A					//Sets PD0-PD7 as I/O pins (Port D)
.equ	PORTB,0x05					//Output I/O Register for port B
.equ	PORTD,0x0B					//Output I/O Registers for port D
.equ	U2X0,1						//Bit position for the USART to double transfer data in UCSR0A
.equ	UBRR0L,0xC4					//Low byte of baud rate register
.equ	UBRR0H,0xC5					//High byte of baud rate register
.equ	UCSR0A,0xC0					//USART control and status registerA
.equ	UCSR0B,0xC1					//USART control register B (enables/disables UART transmitter and receiver)
.equ	UCSR0C,0xC2					//USART control register C (frame format and parity settings)
.equ	UDR0,0xC6					//USART data register for serial data.
.equ	RXC0,0x07					//Bit 7 of UCSR0A for Receive Complete Flag
.equ	UDRE0,0x05					//Bit 5 of UCSR01 for Data register empty flag
.equ	ADCSRA,0x7A					//ADC control and status register A
.equ	ADMUX,0x7C					//ADC multiplexer selection register
.equ	ADCSRB,0x7B					//ADC control and status register B
.equ	DIDR0,0x7E					//Digital input disable register 0 address
.equ	DIDR1,0x7F					//Digital input disable register 1 address
.equ	ADSC,6						//ADC start conversion bit
.equ	ADIF,4						//ADC interrupt flag
.equ	ADCL,0x78					//Address of low byte of ADC data
.equ	ADCH,0x79					//Address of high byte of ADC data
.equ	EECR,0x1F					//EEPROM control register address
.equ	EEDR,0x20					//EEPROM data register address
.equ	EEARL,0x21					//Low byte of EEPROM address
.equ	EEARH,0x22					//High byte of EEPROM address
.equ	EERE,0						//EEPROM read enable bit
.equ	EEPE,1						//EEPROM program enable bit
.equ	EEMPE,2						//EEPROM master write enable
.equ	EERIE,3						//EEPROM ready interrupt enable

.global HADC				//Global high byte of ADC result
.global LADC				//Global low byte of ADC result
.global ASCII				//the memory variable that stores a character to be handled by USART DD
.global DATA				//Shared DATA variable with C for LCD commands

.set	temp,0				//Define symbolic name "temp" for r0

.section ".text"			//Executable program instructions
.global Mega328P_Init
Mega328P_Init:
		;PB0(R*W),PB1(RS),PB2(E) as fixed outputs
		ldi	r16,0x07		//r16 = 0x07
		out	DDRB,r16		//DDRB = r16 = 0x07 = 00000111
						//Each bit represents one of the Port B (PB) registers.
						//This sets this three lowest bits to 1, which correspondingly sets PB0-PB2 as output (p122)
		ldi	r16,0			//r16 = 0
		out	PORTB,r16		//PORTB = r16 = 0
						//Sets all PB registers to driven low (p121)
		;initialize UART, 8bits, no parity, 1 stop, 9600
		out	U2X0,r16		//U2X0 = r16 = 0; sets USART to asynchronous normal mode
		ldi	r17,0x0			//r17 = 0
		ldi	r16,0x67		//r16 = 0x67
		sts	UBRR0H,r17		//UBRR0H = r17 = 0
		sts	UBRR0L,r16		//UBRR0L = r16 = 0x67
						//Combined, this makes UBRR0 0x0067=103.
						//The baud rate formula is 16*10^6/(16(UBRR0+1)). So in this case, the buad rate is 9600. (p264)
		ldi	r16,24			//r16 = 24
		sts	UCSR0B,r16		//UCSR0B = r16 = 24 = 00011000
						//This enables RXEN0 and TXEN0, which is the reciever and transmitter (p285)
		ldi	r16,6			//r16 = 6
		sts	UCSR0C,r16		//UCSR0C = r16 = 6 = 00000110
						//Sets the data size to 8 bits, parity to none, and 1 stop bit (p288)
		
		;initialize ADC - covered in Part 2 by Kang
		ldi	r16,0xE7		//r16 = 0xE7
		sts	ADCSRA,r16		//ADCSRA = r16 = 0xE7 = 11100111. Bit 5, ADATE, is activated. This turns on autotrigger for free running mode
						//Enables ADC and set the ADC division factor to 128 (p354).
		ldi	r16,0x40		//r16 = 0x40
		sts	ADMUX,r16		//ADMUX = r16 = 0x40
		ldi	r16,0			//r16 = 0
		sts	ADCSRB,r16		//ADCSRB = r16 = 0. 0 = 00000000. ADTS[2:0] = 000 = free running
		ldi	r16,0xFE		//r16 = 0xFE
		sts	DIDR0,r16		//DIDR0 = r16 = 0xFE. Disable ADC7-ADC1, keep ADC0 active
		ldi	r16,0xFF		//r16 = 0xFF
		sts 	DIDR1,r16		//DIDR1 = r16 = 0xFF. Disable all bits in DIDR1
		ret				//return
.global UART_Init
UART_Init:
    ; Set asynchronous normal mode (U2X0 = 0)
		ldi     r16,0			//r16 = 0
		out	U2X0,r16		//U2X0 = r16 = 0; sets USART to asynchronous normal mode		
		
    ; Set baud rate to 4800 (UBRR = 207 = 0x00CF)
		ldi     r17,0x00		//r17=0
		ldi     r16,0xCF		//r16=0xCF
		sts     UBRR0H,r17		//UBRR0H = r17 = 0
		sts     UBRR0L,r16		//UBRR0L = r16 = 0xCF
						//Combined, this makes UBRR0 0x00CF = 207.
						//The baud rate formula is 16*10^6/(16(UBRR0+1)). So in this case, the buad rate is 4800. (p264)

    ; Enable RX and TX
		ldi     r16,24			//00011000 (RXEN0 + TXEN0)
		sts     UCSR0B,r16		//UCSR0B = r16 = 24 = 00011000
						//This enables RXEN0 and TXEN0, which is the reciever and transmitter (p285)	

    ; Set frame format: 7 data bits, odd parity, 2 stop bits
		ldi     r16,0x3C		//r16=0x3C=00111100
		sts     UCSR0C,r16		//Sets the data size to 7 bits, parity to odd, and 2 stop bits (p288)

		ret
		
		
    
.global LCD_Write_Command
LCD_Write_Command:
	call		UART_Off		//Disables UART
	ldi		r16,0xFF		//PD0 - PD7 as outputs
	out		DDRD,r16		//Sets DDRD as output for LCD
	lds		r16,DATA		//Apply DATA configuration
	out		PORTD,r16		//Output data to LCD through port D
	ldi		r16,4			//Sets LCD to write
	out		PORTB,r16		//Sends command to LCD
	call		LCD_Delay		//Wait for LCD to process
	ldi		r16,0			//Clear r16 after signal is processed
	out		PORTB,r16		//Complete LCD command
	call		LCD_Delay		//Wait for LCD to process
	call		UART_On			//Re-enable UART
	ret						//Return from subroutine

LCD_Delay:						//Delay routine for LCD to process
	ldi		r16,0xFA			//Outer-loop counter
D0:	ldi		r17,0xFF			//Inner loop counter
D1:	dec		r17				//Decrement inner loop
	brne		D1				//Continue loop until inner counter reaches 0
	dec		r16				//Decrement outer loop
	brne		D0				//Continue delay loop until outer counter reaches 0
	ret						//Return from subroutine, delay is complete

.global LCD_Write_Data
LCD_Write_Data:
	call		UART_Off		//Turn of UART
	ldi		r16,0xFF		//Configure port D as output
	out		DDRD,r16		//Apply output configuration to port D
	lds		r16,DATA		//Load data to be written
	out		PORTD,r16		//Write data to port D
	ldi		r16,6			//Sets LCD controls to write (RS = 1, E = 1, R/w = 0)
	out		PORTB,r16		//Set configuration on port B for LCD controls
	call		LCD_Delay		//Delay for LCD to process
	ldi		r16,0			//Controls control signal
	out		PORTB,r16		//Clears enable lines
	call		LCD_Delay		//Delay for LCD to process
	call		UART_On			//Re-enable UART
	ret					//Return from subroutine

.global LCD_Read_Data
LCD_Read_Data:
	call		UART_Off		//Disable UART
	ldi		r16,0x00		//Configure port D as input
	out		DDRD,r16		//Apply configuration
	out		PORTB,4			//Enable LCD read
	in		r16,PORTD		//Read data from LCD
	sts		DATA,r16		//Store data from LCD into DATA
	out		PORTB,0			//Reset LCD control signal
	call		UART_On			//Re-enable UART
	ret					//Return from subroutine

.global UART_On						//Enable TX/RX
UART_On:
	ldi		r16,2				//r16 = 2
	out		DDRD,r16			//DDRD = r16 = 2 = 00000010
							//Each bit represents one of the Port D (PD) registers.
							//This sets PD1 to output (p128), which is the USART TX
	ldi		r16,24				//r16 = 24
	sts		UCSR0B,r16			//UCSR0B = r16 = 24 = 00011000
							//This enables RXEN0 and TXEN0, which is the reciever and transmitter (p285)
	ret						//return

.global UART_Off					//Disable TX/RX
UART_Off:
	ldi	r16,0					//r16 = 0
	sts	UCSR0B,r16				//UCSR0B = r16 = 0
							//This disables RXEN0 and TXEN0, which is the reciever and transmitter (p285)
	ret						//return

.global UART_Clear					//Empty the USART buffer
UART_Clear:
	lds		r16,UCSR0A			//r16 = UCSR0A
	sbrs		r16,RXC0			//Skip the return if the RXC bit in UCSR0A is tripped.
							//Which means, return unless there is unread data in USART buffer (p283)
	ret						//Return only if no unread data in USART buffer
	lds		r16,UDR0			//r16 = UDR0 (which is the USART buffer)
	rjmp		UART_Clear			//call this instruction again after clearing USART buffer

.global UART_Get					//Wait for incoming USART data
UART_Get:
	lds		r16,UCSR0A			//r16 = UCSR0A
	sbrs		r16,RXC0			//Skip the rjmp if the RXC bit in UCSR0A is tripped.
							//Which means, jump unless there is unread data in USART buffer (p283)
	rjmp		UART_Get			//call this instruction again if there is no unread data in USART buffer
	lds		r16,UDR0			//r16 = UDR0 (which is the USART buffer)
	sts		ASCII,r16			//ASCII = r16 (ASCII is a character to be handled by USART)
	ret						//return

.global UART_Put					//Send data
UART_Put:
	lds		r17,UCSR0A			//r17 = UCSR0A
	sbrs		r17,UDRE0			//Skip the rjmp if UDRE0 bit in UCSR0A is tripped.
							//Which means, jump unless the USART buffer is empty (p283)
	rjmp		UART_Put			//call this instruction again if the USART buffer has data in it
	lds		r16,ASCII			//r16 = ASCII (ASCII is a character to be handled by USART)
	sts		UDR0,r16			//UDR0 = r16
	ret						//return

.global ADC_Get
ADC_Get:
		lds		r16,ADCL		//Reads low byte of ADC
		sts		LADC,r16		//Loads value in ADC Low Register
		lds		r16,ADCH		//Reads High Byte of ADC
		sts		HADC,r16		//Loads value into ADC High register
		ret						//Return to ADC_Get
								//Now this function will keep running to read the current volatge value

.global EEPROM_Write
EEPROM_Write:      
		sbic		EECR,EEPE
		rjmp		EEPROM_Write			; Wait for completion of previous write
		ldi		r18,0x00			; Set up address (r18:r17) in address register
		ldi		r17,0x05 
		ldi		r16,'F'				; Set up data in r16    
		out		EEARH, r18      
		out		EEARL, r17			      
		out		EEDR,r16			; Write data (r16) to Data Register  
		sbi		EECR,EEMPE			; Write logical one to EEMPE
		sbi		EECR,EEPE			; Start eeprom write by setting EEPE
		ret 

.global EEPROM_Read
EEPROM_Read:					    
		sbic		EECR,EEPE    
		rjmp		EEPROM_Read		; Wait for completion of previous write
		ldi		r18,0x00		; Set up address (r18:r17) in EEPROM address register
		ldi		r17,0x05
		ldi		r16,0x00   
		out		EEARH, r18   
		out		EEARL, r17		   
		sbi		EECR,EERE		; Start eeprom read by writing EERE
		in		r16,EEDR		; Read data from Data Register
		sts		ASCII,r16  
		ret


		.end
		
.section .eeprom
InitialByte:
